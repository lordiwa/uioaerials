import { gsap } from 'gsap'
import { ScrollTrigger } from 'gsap/ScrollTrigger'
import { onMounted, onUnmounted } from 'vue'

gsap.registerPlugin(ScrollTrigger)

export function useReveal(selector: string, options = {}) {
  const defaults = {
    y: 40, opacity: 0, duration: 0.9,
    ease: 'power3.out', stagger: 0.1,
    scrollTrigger: { start: 'top 85%', toggleActions: 'play none none none' }
  }
  let ctx: gsap.Context

  onMounted(() => {
    ctx = gsap.context(() => {
      gsap.from(selector, { ...defaults, ...options })
    })
  })
  onUnmounted(() => ctx?.revert())
}

import Lenis from 'lenis'

let lenis: Lenis

export function initLenis() {
  lenis = new Lenis({ lerp: 0.08, smooth: true })
  lenis.on('scroll', ScrollTrigger.update)

  gsap.ticker.lagSmoothing(0)
}
